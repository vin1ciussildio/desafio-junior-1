Banco de dados:


create database petshop;

create table dono (
cpf varchar (30) not null primary key,
nome varchar (80), 
idade int,
endereco tinytext,
email varchar (80),
cell varchar (30)
);
drop table dono;

create table animal (
id int not null auto_increment primary key ,
nome varchar(30),
tipo varchar(30),
idade int,
cpf_dono varchar (30) 
);

alter table animal add foreign key (cpf_dono) references dono (cpf);
